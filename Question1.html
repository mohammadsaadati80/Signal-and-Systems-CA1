import numpy as np
from matplotlib import pyplot as plt

def x(A, n, k):
    x = A*(np.cos((2*np.pi)* k * n))
    return x

def X(n):
    y = x(A, n, k)
    plt.stem(n, y)
    plt.title('X[n]')
    plt.xlabel('n')
    plt.ylabel('X[n]')
    plt.show()
